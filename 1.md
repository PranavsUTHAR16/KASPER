graph TD
    %% Setup and Initialization
    subgraph Setup ["1. Initialization & Setup"]
        A1[Load Split Datasets: train, val] --> A2[Fit Target StandardScaler on train targets]
        A2 --> A3[Instantiate KASPER & fit_knots on training features]
        A3 --> A4[Instantiate KasperCompositeLoss]
        A4 --> A5[Configure AdamW Optimizer & ReduceLROnPlateau Scheduler]
    end

    %% Epoch loop
    subgraph EpochLoop ["2. Epoch Loop (Epoch t = 1..100)"]
        B1[Calculate Current Gumbel Temperature: tau_t] --> B2[Set Model to Train Mode: model.train]
    end
    Setup --> EpochLoop

    %% Batch processing
    subgraph BatchLoop ["3. Batch Training Loop"]
        C1[Fetch Batch: x_batch, y_batch] --> C2[Model Forward Pass: prediction, probs, embeddings]
        C2 --> C3[Calculate Huber, Sparsity, Contrastive, Orthogonality, Balance Losses]
        C3 --> C4[Backward Pass: loss.backward]
        C4 --> C5[Check Layer 1 Projection Gradient Norm]
        C5 --> C6[Gradient Clipping: max_norm = 0.5]
        C6 --> C7[Optimizer Step: optimizer.step]
    end
    EpochLoop --> BatchLoop
    C7 -- Next Batch --> C1

    %% Epoch End Hooks
    subgraph EpochEnd ["4. Epoch-End Diagnostics & Validation"]
        D1[Epoch Sparsity Monitor Hook: Print pruned weight percentage] --> D2[Validation Pass: Calculate Val Loss]
        D2 --> D3[ReduceLROnPlateau Scheduler Step]
    end
    BatchLoop -- Batch Loop Finished --> EpochEnd

    %% Early Stopping
    subgraph Decision ["5. Optimization & Save Decision"]
        E1{Val Loss Improved?}
        E2[Save weights to best_kasper.pth]
        E3[Increment early stopping patience counter]
        E4{Patience >= 15?}
        E5[Terminate Training & Save Best Model]
        
        E1 -- Yes --> E2
        E1 -- No --> E3
        E3 --> E4
        E4 -- Yes --> E5
        E4 -- No --> F1[Continue to Epoch t+1]
    end
    EpochEnd --> Decision
    F1 --> EpochLoop
